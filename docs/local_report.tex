\section{Configuração}
\begin{itemize}
  \item Baseline: $[-2.0, -1.0]$ s (auto-referência C3/C4, combinaçao ótima)
  \item Janela de detecção: $[0, 4]$ s
  \item Limiar: $-0{,}60\,\sigma$ (ajustado)
  \item Mínimo de canais: 1 (C3 ou C4)
  \item Artefatos: limiar 750 $\mu$V
  \item Seleção de IMF: potência $\geq 35$ por cento em 8--30 Hz
\end{itemize}

\section{Interpretação Rápida}
\begin{itemize}
\item A taxa de detecção reflete a configuração sensível (1 canal, -0{,}60$\sigma$), favorecendo alta sensibilidade.
\item O baseline ([-2,-1] s) equilibra detecção precoce e estabilidade.
\item Ensaios com artefatos (picos $>$ 750 $\mu$V) permanecem contabilizados; considere filtragem adicional para análise final.
\end{itemize}
